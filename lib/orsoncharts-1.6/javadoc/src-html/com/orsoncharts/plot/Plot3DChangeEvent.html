033package com.orsoncharts.plot;
034
035import java.util.EventObject;
036
037import com.orsoncharts.Chart3D;
038import com.orsoncharts.data.Dataset3DChangeEvent;
039import com.orsoncharts.util.ArgChecks;
040
041/**
042 * An event used to signal a change to a {@link Plot3D}.  Any object that
043 * implements the {@link Plot3DChangeListener} interface can register with a 
044 * plot to receive change notifications.  By default, the {@link Chart3D}
045 * object will register with the plot it manages to monitor changes to the plot
046 * and its subcomponents.
047 * <br><br>
048 * NOTE: This class is serializable, but the serialization format is subject 
049 * to change in future releases and should not be relied upon for persisting 
050 * instances of this class. 
051 */
052@SuppressWarnings("serial")
053public class Plot3DChangeEvent extends EventObject {
054
055    /** The plot. */
056    private Plot3D plot;
057    
058    /** Does the plot change require the world to be updated? */
059    private boolean requiresWorldUpdate;
060  
061    /**
062     * Creates a new event.  The {@code source} of the event can be
063     * either the plot instance or another event that was received by the
064     * plot (for example, a {@link Dataset3DChangeEvent}).
065     * 
066     * @param source  the event source ({@code null} not permitted).
067     * @param plot  the plot ({@code null} not permitted).
068     * @param requiresWorldUpdate  a flag that indicates whether or not the 
069     *     world requires updating because of this change.
070     */
071    public Plot3DChangeEvent(Object source, Plot3D plot, 
072            boolean requiresWorldUpdate) {
073        super(source);
074        ArgChecks.nullNotPermitted(plot, "plot");
075        this.plot = plot;
076        this.requiresWorldUpdate = requiresWorldUpdate;
077    }
078 
079    /**
080     * Returns the plot from which the event came.
081     * 
082     * @return The plot (never {@code null}). 
083     */
084    public Plot3D getPlot() {
085        return this.plot;
086    }
087    
088    /**
089     * Returns the flag indicating whether or not this change event
090     * requires the world to be updated/recreated.
091     * 
092     * @return A boolean.
093     * 
094     * @since 1.2
095     */
096    public boolean requiresWorldUpdate() {
097        return this.requiresWorldUpdate;
098    }
099    
100}
