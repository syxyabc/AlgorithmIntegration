033package com.orsoncharts.marker;
034
035import java.awt.BasicStroke;
036import java.awt.Color;
037import java.awt.Font;
038import java.awt.Graphics2D;
039import java.awt.Stroke;
040import com.orsoncharts.ChartElement;
041
042/**
043 * A base interface for all markers.
044 * 
045 * @since 1.2
046 */
047public interface Marker extends ChartElement {
048
049    /** The default line stroke for markers. */
050    public static final Stroke DEFAULT_LINE_STROKE = new BasicStroke(0f, 
051            BasicStroke.CAP_ROUND, BasicStroke.JOIN_ROUND);
052
053    /** The default line color for markers. */
054    public static final Color DEFAULT_LINE_COLOR = Color.DARK_GRAY;
055
056    /** The default fill color for markers. */
057    public static final Color DEFAULT_FILL_COLOR = new Color(128, 128, 192, 64);
058
059    /** The default font for marker labels. */
060    public static final Font DEFAULT_MARKER_FONT = new Font(Font.DIALOG, 
061            Font.PLAIN, 10);
062    
063    /** The default color for the marker labels. */
064    public static final Color DEFAULT_LABEL_COLOR = Color.BLACK;
065    
066    /** 
067     * Draws the marker based on the {@code markerData} which has been 
068     * passed to the 3D engine to generate the required 2D projection points.
069     * 
070     * @param g2  the graphics target ({@code null} not permitted).
071     * @param markerData  transient data for the marker ({@code null} not 
072     *     permitted).
073     * @param reverse  a flag to indicate reverse orientation.
074     */
075    void draw(Graphics2D g2, MarkerData markerData, boolean reverse);
076    
077    /**
078     * Registers a listener to receive notification of changes to the marker.
079     * 
080     * @param listener  the listener ({@code null} not permitted). 
081     */
082    void addChangeListener(MarkerChangeListener listener);
083    
084    /**
085     * Deregisters a listener so that it no longer receives notification of 
086     * changes to the marker.
087     * 
088     * @param listener  the listener ({@code null} not permitted). 
089     */
090    void removeChangeListener(MarkerChangeListener listener);
091    
092}
